# Database Backup Schedule for ContentFlow SaaS Platform
# This file should be added to the system crontab

# Daily full backup at 2 AM
0 2 * * * /bin/bash /app/scripts/backup-database.sh full

# Incremental backup every 6 hours
0 */6 * * * /bin/bash /app/scripts/backup-database.sh incremental

# Transaction log backup every 15 minutes
*/15 * * * * /bin/bash /app/scripts/backup-database.sh transaction-log

# Backup validation every day at 3 AM
0 3 * * * /bin/bash /app/scripts/validate-backup.sh /backups/latest-full-backup.sql.gz

# Migration monitoring every hour
0 * * * * /bin/bash /app/scripts/monitor-migration.sh

# Backup monitoring every hour
0 * * * * /bin/bash /app/scripts/monitor-backup.sh
